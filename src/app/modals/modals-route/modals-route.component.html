Modals

A simple directive to apply to an ng-template containing some Bootstrap modal markup.

API

Module: ModalModule

import {{ '{' }} ModalModule {{ '}' }} from '@nowzoo/ngx-strap';

Directive: ModalDirective

selector: '[ngxStrapModal]', exportAs: 'ngxStrapModal'

Inputs

none

Method

show(): IModalInstance

Interface: IModalInstance

events: EventEmitter<Event>;
